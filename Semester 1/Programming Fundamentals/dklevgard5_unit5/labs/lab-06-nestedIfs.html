// YOUR SOLUTION ALGORITHM GOES AFTER THIS LINE:
KidsAndRooms
  Prompt for kids 
  Prompt for bedrooms
  
  IF  kids == 1 THEN
    IF bedrooms == 1 THEN
      Display "With 1 kid and 1 bedroom you must be feel lucky!"
    IF bedrooms == 2 THEN
      Display "With 1 kid and 2 bedrooms it must feel spacious!"
    IF bedrooms >= 3  THEN 
      Display " With 1 kid and " + bedrooms + " bedrooms you must have an empty feeling!"
  ELSE IF kids == 2 THEN 
    IF bedrooms == 1 THEN
      Display "With 2 kids and 1 bedroom you must be feel cozy!"
    IF bedrooms == 2 THEN
      Display "With 2 kids and 2 bedrooms you must feel lucky!"
    IF bedrooms >= 3  THEN 
      Display " With 2 kids and " + bedrooms + " bedrooms it must feel spacious!"
   ELSE IF kids >= 3 THEN 
    IF bedrooms == 1 THEN
      Display "With 3 kids and 1 bedroom you must be feel crowded!"
    IF bedrooms == 2 THEN
      Display "With 3 kids and 2 bedrooms you must feel cozy!"
    IF bedrooms >= 3  THEN 
      Display " With 3 kids and " + bedrooms + " bedrooms you must feel lucky!"
  ENDIF
END
